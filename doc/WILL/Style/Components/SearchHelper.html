    ws_search(path="", query="", placeholder="Search...", options={}, &block) click to toggle source

    Generates a search bar. This returns a search builder that you can use to fill in a custom options menu if you want. The search bar and button are generated for you automatically.

    Params¶ ↑

    path

    The search path for the search form.

    +query (optional)+

    The existing search query for the search form

    placeholder

    Placeholder text for an empty search field

def ws_search(path="", query="", placeholder="Search...", options={}, &block)
  # Make query and placeholder completely optional
  options = placeholder if placeholder.is_a?(Hash)
  options = query if query.is_a?(Hash)

  # Set up mandatory options
  options[:class] = "search-form #{options[:class]}"
  options[:action] = "search"
  options[:method] = :get
  options[:enforce_utf8] = false

  form_tag(path, options) do
    builder = SearchBuilder.new(self, query)
    block_content = block_given? ? capture(builder, &block) : ""

    content_tag(:div, class: "row collapse") do
      [
        block_content,                
        builder.input(placeholder),
        builder.button,
        utf8_enforcer_tag
      ].join.html_safe
    end
  end
end
